import java.util.Iterator;
public class ArraySet implements Iterable {
	
	private class ArraySetIterator implements Iterator {
		private int current = 0;

        @Override
        public boolean hasNext() {
            return this.current < ArraySet.this.size;
        }

        @Override
        public T next() {
            T t = ArraySet.this.data[this.current];
            this.current += 1;
            return t;
        }

        @Override
        public void remove() {
            throw new UnsupportedOperationException();
        }
	}
	
	private int size;
	private T[] data;
	
	public ArraySet() {
		this.data = (T[]) new Object[1];
	}
	
	//called when add is called on set and they
	//underlying array is full and needs to be 
	//doubled in size
	private void grow() {
		T[] newArray = (T[]) new Object[this.data.length * 2];
		for (int i = 0; i < this.data.length; i++) {
			newArray[i] = this.data[i];
		}
		this.data = newArray;
	}
	
	//returns the position int the array that
	//t is in, returns -1 if not found.
	private int find(T t) {
		for (int i = 0; i < this.size; i++) {
			if (this.data[i].equals(t)) {
				return i;
			}
		}
		return -1;
	}
	
	public void add(T t) {
		if (this.find(t) != -1) {
			System.out.println("Element already in set");
			//can throw your own exception here
			return;
		} else if (this.size == this.data.length) {
			this.grow();
		}
		this.data[this.size] = t;
		this.size++;
	}
	
	public void remove(T t) {
		int index = this.find(t);
		if (index == -1) {
			System.out.println("Element not in set");
			//can throw your own exception here
			return;
		}
		T temp = this.data[index];
		this.data[index] = this.data[this.size - 1];
		this.data[size - 1] = temp;
		this.size--;
	}
	
	public boolean has(T t) {
		return this.find(t) != -1;
	}
	
	public int size() {
		return this.size;
	}
	
	public Iterator iterator() {
		return new ArraySetIterator();
	}
}
