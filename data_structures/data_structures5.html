public class ArrayQueue {
	
	private int front;
	private int back;
	private int size;
	private T[] data;
	
	public ArrayQueue() {
		this.size = 0;
		this.front = 0;
		this.back = 0;
		this.data = (T[]) new Object[1];
	}
	
	public void add(T t) {
		if (this.size == this.data.length) {
			this.grow();
		}
		
		if (back == data.length - 1) {
			back = 0;
		} else {
			back++;
		}
		data[back] = t;
		this.size++;
	}
	
	public T peek() {
		if (this.size == 0) {
			System.out.println("queue is empty");
			//can insert custom exception here
			return null;
		}
		return this.data[front];
	}
	
	public T remove() {
		if (this.size == 0) {
			System.out.println("queue is empty");
			//can insert custom exception here
			return null;
		}
		
		T returnVal = this.data[front];
		if (this.front == this.data.length - 1) {
			this.front = 0;
		} else {
			this.front++;
		}
		this.size--;
		return returnVal;
	}
	
	private void grow() {
		//make the new array with double the size
		T[] newArray = (T[]) new Object[data.length * 2];
		int copyIndex = 0;
		int i = front;
		
		//copy elements from front to back
		while (copyIndex < this.size()) {
			newArray[copyIndex] = data[i];
			if (i == this.data.length - 1) {
				i = 0;
			} else {
				i++;
			}
			copyIndex++;
		}
		
		//reassign the new array and indexes.
		this.data = newArray;
		front = 0;
		back = this.size - 1;
	}
	
	public int size() {
		return size;
	}
	
	public boolean isEmpty() {
		return this.size == 0;
	}
	
}

import java.util.LinkedList;

public class LinkedDeque {
	
	private LinkedList data;
	private int size;
	
	public LinkedDeque() {
		this.size = 0;
		this.data = new LinkedList();
	}
	
	public T peekFront() {
		if (this.size == 0) {
			System.out.println("Deque Empty");
			//you can add custom exception.
			return null;
		}
		return this.data.peekFirst();
	}
	
	public T peekBack() {
		if (this.size == 0) {
			System.out.println("Deque Empty");
			//you can add custom exception.
			return null;
		}
		return this.data.peekLast();
	}
	
	public void addFront(T t) {
		this.data.addFirst(t);
		this.size++;
	}
	
	public void addBack(T t) {
		this.data.addLast(t);
		this.size++;
	}
	
	public T removeFront() {
		if (this.size == 0) {
			System.out.println("Deque Empty");
			//you can add custom exception.
			return null;
		}
		this.size--;
		return this.data.removeFirst();
	}
	
	public T removeBack() {
		if (this.size == 0) {
			System.out.println("Deque Empty");
			//you can add custom exception.
			return null;
		}
		this.size--;
		return this.data.removeLast();
	}
	
	public int size() {
		return this.size;
	}
	
	public boolean isEmpty() {
		return this.size == 0;
	}
}
