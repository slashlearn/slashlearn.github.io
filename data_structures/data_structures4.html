import java.util.LinkedList;

public class LinkedListStack {
	
	private LinkedList data;
	private int size;
	
	public LinkedListStack() {
		this.data = new LinkedList();
		this.size = 0;
	}
	
	public void push(T t) {
		this.data.addFirst(t);
		this.size++;
	}
	
	public T peek() {
		if (this.size == 0) {
			//can throw exception if you want
			System.out.println("Stack is empty");
			return null;
		}
		return this.data.getFirst();
	}
	
	public T pop() {
		if (this.size == 0) {
			//can throw exception if you want
			System.out.println("Stack is empty");
			return null;
		}
		size--;
		return this.data.removeFirst();
	}
	
	public int size() {
		return this.size;
	}
	
	public boolean isEmpty() {
		return this.size == 0;
	}
	
}

public class ArrayStack {
	
	private T[] data;
	private int topIdx;
	
	public ArrayStack() {
		this.data = (T[]) new Object[2];
		this.topIdx = -1;
	}
	
	private void resize() {
		T[] newData = (T[]) new Object[this.data.length * 2];
        for (int i = 0; i < this.topIdx + 1; i++) {
            newData[i] = this.data[i];
        }
        this.data = newData;
	}
	
	public void push(T t) {
		if (topIdx == data.length - 1)
			this.resize();
		this.data[this.topIdx + 1] = t;
		topIdx++;
	}
	
	public T peek() {
		if (this.size() == 0) {
			//can throw exception if you want
			System.out.println("Stack is empty");
			return null;
		}
		return this.data[topIdx];
	}
	
	public T pop() {
		if (this.size() == 0) {
			//can throw exception if you want
			System.out.println("Stack is empty");
			return null;
		}
		this.topIdx--;
		return this.data[topIdx + 1];
	}
	
	public int size() {
		return this.topIdx + 1;
	}
	
	public boolean isEmpty() {
		return this.topIdx == -1;
	}
	
}
