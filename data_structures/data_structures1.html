import java.util.Iterator;
/**
 * Interface for our own Array Data Structure
 * @param  The Base type for an array.
 */
public interface Array {
	
	/**
	 * The maximum size of the array.
	 * @return the length of the array.
	 */
	int length();
	
	/**
	 * Gets the value of type T at a certain index.
	 * @param i the index of the value.
	 * @return the value at index i
	 */
	T get(int i);
	
	/**
	 * Sets the value at position i to t.
	 * @param i index of data to be set.
	 * @param t what the value at i will be set to.
	 */
	void set(int i, T t);

	/**
	 * Creates an iterable object for Array.
	 * @return the iterable object.
	 */
	Iterator iterator();
}

Code Segment: BasicArray.java 
import java.util.Iterator;
public class BasicArray implements Array, Iterable {

    private class BasicArrayIterator implements Iterator {
        
        int current = 0; //current position for iterator

        @Override
        public T next() {
            T t = BasicArray.this.data[this.current];
            this.current += 1;
            return t;
        }

        @Override
        public boolean hasNext() {
            return this.current < BasicArray.this.data.length;
        }

        @Override
        public void remove() {
            throw new UnsupportedOperationException();
        }
    }
    
    private T[] data;
	
	/**
	 * Creates and returns new array of i length that holds 
	 * values of the type T. 
	 * Initial value of array is t.
	 * @param t the initial value for all positions in the array.
	 * @param length the maximum capacity of our array.
	 */
	public BasicArray(T t, int length) {
		data = (T[]) new Object[length];
		for (int i = 0; i < length; i++) {
			data[i] = t;
		}
	}
	
    @Override
    public Iterator iterator() {
        return new BasicArrayIterator();
    }
    
	public int length() {
		return data.length;
	}
	
	public void set(int i, T t) {
		if (checkValidIndex(i)) {
			data[i] = t;
		} else {
			throw new ArrayIndexOutOfBoundsException();
		}
	}
	
	public T get(int i) {
		if (!checkValidIndex(i))
			throw new ArrayIndexOutOfBoundsException();
		return data[i];
	}
	
	private boolean checkValidIndex(int i) {
		return (i >= 0) && (i < data.length);
	}
}

Code Segment: Tester.java 
import java.util.Iterator;
public class Tester {

	public static void main(String[] args) {
		BasicArray arr = new BasicArray(0, 5);
		
		for (int i = 0; i < arr.length(); i++) {
			arr.set(i, i * 10);
		}

		for (Integer i : arr) {
			System.out.print(i + ", ");
		}
		System.out.println();
		
		Iterator it = arr.iterator();
		while (it.hasNext()) {
			Integer i = it.next();
			System.out.print(i + ", ");
		}
	}
}
