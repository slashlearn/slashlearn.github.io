import java.util.Iterator;
import java.util.NoSuchElementException;
public class DoublyLinkedList implements Iterable{
        
    private class Node {
        T data;
        Node next;
        Node prev;
                
        public Node(T t) {
            this.data = t;
            this.next = null;
        }       
    }
    
    private class ListIterator implements Iterator {
        Node current;
        boolean forward;

        ListIterator(boolean f) {
            this.forward = f;
            if (this.forward) {
                this.current = DoublyLinkedList.this.sentinelHead.next;
            } else {
                this.current = DoublyLinkedList.this.sentinelTail.prev;
            }
        }

        @Override
        public T next() throws {
            if (!this.hasNext()) {
                throw new NoSuchElementException();
            }
            T t = this.current.data;
            if (this.forward) {
                this.current = this.current.next;
            } else {
                this.current = this.current.prev;
            }
            return t;
        }

        @Override
        public boolean hasNext() {
            if (this.forward) {
                return this.current != DoublyLinkedList.this.sentinelTail;
            }
            return this.current != DoublyLinkedList.this.sentinelHead;
        }

        @Override
        public void remove() {
            throw new UnsupportedOperationException();
        }
    }
        
    private int length;
    private Node sentinelHead;
    private Node sentinelTail;
        
    public DoublyLinkedList() {
        this.sentinelHead = new Node<>(null);
        this.sentinelTail = new Node<>(null);
        this.sentinelHead.next = this.sentinelTail;
        this.sentinelTail.prev = this.sentinelHead;
        this.length = 0;
    }
    
    public boolean insertAfter(T key, T t) {
        Node current = sentinelHead;
        while (current != key) {
            if (current == this.sentinelTail)
                return false;
            else if(key.equals(current.data))
                break;
            current = current.next;
        }
        Node n = new Node<>(t);
        n.next = current.next;
        n.next.prev = n;
        current.next = n;
        n.prev = current;
        this.length++;
        return true;
    }
    
    public boolean insertBefore(T key, T t) {
        Node current = sentinelTail;
        while (current != key) {
            if (current == this.sentinelHead)
                return false;
            else if(key.equals(current.data))
                break;
            current = current.prev;
        }
        Node n = new Node<>(t);
        n.prev = current.prev;
        n.prev.next = n;
        n.next = current;
        current.prev = n;
        this.length++;
        return true;
    }
    
    public void insertHead(T t) {
        Node n = new Node<>(t);
        n.next = this.sentinelHead.next;
        n.next.prev = n;
        n.prev = this.sentinelHead;
        this.sentinelHead.next = n;
        this.length++;
    }
    
    public void insertTail(T t) {
        Node n = new Node<>(t);
        
    }
    
    public int length() {
        return this.length;
    }
    
    public boolean isEmpty() {
        return this.length == 0;
    }
    
    public T deleteVal(T key) {
        if (this.isEmpty())
            throw new NullPointerException();
        //we never delete a sentinel Node
        Node current = sentinelHead.next;
        while (current != key) {
            if (current == this.sentinelTail)
                return null;
            else if(key.equals(current.data))
                break;
            current = current.next;
        }
        current.prev.next = current.next;
        current.next.prev = current.prev;
        this.length--;
        return current.data;
    }
    
    public T deleteHead() {
        if (this.isEmpty())
            throw new NullPointerException();
        Node temp = this.sentinelHead.next;
        this.sentinelHead.next = temp.next;
        temp.next.prev = this.sentinelTail;
        this.length--;
        return null;
    }
    
    public T deleteTail() {
        if (this.isEmpty()) 
            throw new NullPointerException();
        //temp is the node that will be deleted
        Node temp = this.sentinelTail.prev;
        this.sentinelTail.prev = temp.prev;
        temp.prev.next = this.sentinelTail;
        this.length--;
        return temp.data;
    }
    
    public T getHead() {
        return this.sentinelHead.next.data;
    }
    
    public T getTail() {
        return this.sentinelTail.next.data;
    }
    
    //forward iterator
    public Iterator forward() {
        return new ListIterator(true);
    }

    //backward iterator
    public Iterator backward() {
        return new ListIterator(false);
    }

    public Iterator iterator() {
        return this.forward();
    }
    
}
